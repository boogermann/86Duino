
# AVR compile variables
# --------------------- 

name=Vortex86EX (32-bits) Boards
# Default "compiler.path" is correct, change only if you want to overidde the initial value
upx.path={runtime.ide.path}\hardware\tools\upx309w\
compiler.path={runtime.ide.path}\hardware\tools\djgpp\bin\
compiler.lib.path={runtime.ide.path}\hardware\tools\djgpp\lib\
compiler.include.path={runtime.ide.path}\hardware\tools\djgpp\i586-pc-msdosdjgpp\sys-include\
compiler.c.cmd=i586-pc-msdosdjgpp-gcc
compiler.c.flags=-c -w -g -MMD -ffunction-sections -fdata-sections -fpermissive -O1  
compiler.c.elf.flags=-w -Wl,--gc-sections -O1
compiler.c.elf.cmd=i586-pc-msdosdjgpp-gcc
compiler.S.flags=-c -g -assembler-with-cpp
compiler.cpp.cmd=i586-pc-msdosdjgpp-gcc
compiler.cpp.flags=-c -w -g -MMD -fno-exceptions -ffunction-sections -fdata-sections -fno-threadsafe-statics -fpermissive -O1  
compiler.ar.cmd=i586-pc-msdosdjgpp-ar
compiler.ar.flags=rcs
compiler.objcopy.cmd=avr-objcopy
compiler.objcopy.eep.flags=-O ihex -j .eeprom --set-section-flags=.eeprom=alloc,load --no-change-warnings --change-section-lma .eeprom=0
compiler.elf2hex.flags=-v
compiler.elf2hex.cmd=i586-pc-msdosdjgpp-strip
compiler.upx.cmd=upx
compiler.ldflags=
compiler.size.cmd=i586-pc-msdosdjgpp-size
# this can be overriden in boards.txt
build.extra_flags=

# AVR compile patterns
# --------------------

## Compile c files
recipe.c.o.pattern="{compiler.path}{compiler.c.cmd}" {compiler.c.flags} -DARDUINO={runtime.ide.version} -D_86DUINO={runtime.ide86.version} "{source_file}" -o "{object_file}" {includes} "-I{compiler.lib.path}" "-I{compiler.include.path}"

## Compile c++ files
recipe.cpp.o.pattern="{compiler.path}{compiler.cpp.cmd}" {compiler.cpp.flags} -DARDUINO={runtime.ide.version} -D_86DUINO={runtime.ide86.version} "{source_file}" -o "{object_file}" {includes} "-I{compiler.lib.path}" "-I{compiler.include.path}"

## Create archives
recipe.ar.pattern="{compiler.path}{compiler.ar.cmd}" {compiler.ar.flags} "{build.path}/{archive_file}" "{object_file}" 

## Combine gc-sections, archives, and objects
recipe.c.combine.pattern="{compiler.path}{compiler.c.elf.cmd}" {compiler.c.elf.flags} -o "{build.path}/86duino.exe" {object_files} "-L{build.path}" "-L{compiler.lib.path}" -lcore {str_libfiles} -lstdc++

## Create eeprom
recipe.objcopy.eep.pattern="{compiler.path}{compiler.objcopy.cmd}" {compiler.objcopy.eep.flags} "{build.path}/{build.project_name}.elf" "{build.path}/{build.project_name}.eep"

## Create hex
recipe.objcopy.hex.pattern="{compiler.path}{compiler.elf2hex.cmd}" {compiler.elf2hex.flags} "{build.path}/86duino.exe"

## use upx
recipe.objcopy.upx.pattern="{upx.path}{compiler.upx.cmd}" "{build.path}/86duino.exe"  

## Compute size
recipe.size.pattern="{compiler.path}{compiler.size.cmd}" -A "{build.path}/86duino.exe"
recipe.size.regex=Total\s+([0-9]+).*
